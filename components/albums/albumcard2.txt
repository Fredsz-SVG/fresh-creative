function AlbumCard({
  album,
  variant,
  basePath,
  pathname,
  onApprove,
  onDecline,
  onDelete,
  onInvite,
  onPay,
  loadingId,
  navigatingAlbumId,
  setNavigatingAlbumId,
  onYearbookPublicChoice,
}: {
  album: AlbumRow
  variant: 'user' | 'admin'
  basePath: string
  pathname: string | null
  onApprove?: (album: AlbumRow) => void
  onDecline?: (album: AlbumRow) => void
  onDelete?: (album: AlbumRow) => void
  onInvite?: (album: AlbumRow) => void
  onPay?: (album: AlbumRow) => void
  loadingId?: string | null
  navigatingAlbumId?: string | null
  setNavigatingAlbumId?: (id: string | null) => void
  onYearbookPublicChoice?: (album: AlbumRow) => void
}) {
  const router = useRouter()
  const isAdmin = variant === 'admin'
  const isPaid = album.isOwner !== false ? album.payment_status === 'paid' : (album.individual_payments_enabled ? album.member_payment_status === 'paid' : album.payment_status === 'paid')
  const isApproved = album.status === 'approved'
  const isClickable = album.type === 'public' || (isApproved && (isPaid || isAdmin))

  const destinationUrl = album.type === 'public'
    ? `${basePath}/album/public/${album.id}`
    : `/album/${album.album_id ?? album.id}/view`

  const editorUrl = album.type === 'yearbook'
    ? `${basePath}/album/yearbook/${album.album_id ?? album.id}`
    : null

  const created = album.created_at ? new Date(album.created_at).toLocaleDateString('id-ID', { day: 'numeric', month: 'short', year: 'numeric' }) : null
  const statusLabel = album.type === 'yearbook' ? (album.status ?? 'pending') : 'public'
  const canSeeApproved = isAdmin || album.isOwner === true
  const shouldShowStatus = !(statusLabel === 'approved' && !canSeeApproved)
  const displayStatus = statusLabel as string
  const isLoading = loadingId === album.id
  const isNavigatingToEditor = navigatingAlbumId === album.id

  const CardContent = () => (
    <div
      onClick={(e) => {
        if (isClickable && !isLoading) {
          try { router.prefetch(destinationUrl) } catch { }
          if (album.type === 'yearbook' && onYearbookPublicChoice) {
            onYearbookPublicChoice(album)
            return
          }
          router.push(destinationUrl, { scroll: false })
        }
      }}
      onMouseEnter={() => {
        if (!isClickable) return
        try { router.prefetch(destinationUrl) } catch { }
      }}
      onTouchStart={() => {
        if (!isClickable) return
        try { router.prefetch(destinationUrl) } catch { }
      }}
      className={`group relative h-full flex flex-col bg-white dark:bg-slate-900 rounded-2xl overflow-hidden transition-all duration-200 border-2 border-slate-200 dark:border-slate-700 ${
        isClickable
          ? 'cursor-pointer shadow-[4px_4px_0_0_#334155] dark:shadow-[4px_4px_0_0_#1e293b] hover:shadow-none hover:-translate-x-0.5 hover:-translate-y-0.5'
          : 'cursor-default opacity-70'
      }`}
    >
      <div className="relative aspect-[3/4] bg-gradient-to-br from-pink-100 to-violet-100 dark:from-pink-950/40 dark:to-violet-950/40 overflow-hidden">
        {album.cover_image_url ? (
          <FastImage
            src={album.cover_image_url}
            alt={album.name}
            className="w-full h-full object-cover"
            style={album.cover_image_position ? { objectPosition: album.cover_image_position } : undefined}
          />
        ) : (
          <div className="w-full h-full flex flex-col items-center justify-center gap-3 text-slate-400 dark:text-slate-500">
            <BookOpen className="w-12 h-12" />
            <span className="text-xs font-semibold uppercase tracking-widest">No Cover</span>
          </div>
        )}

        {album.type === 'public' && (
          <div className="absolute top-3 left-3 px-2.5 py-1 bg-indigo-500 text-white text-[10px] font-bold uppercase tracking-wider rounded-full">
            Public
          </div>
        )}

        <div className="absolute inset-0 bg-gradient-to-t from-black/50 via-transparent to-transparent opacity-0 group-hover:opacity-100 transition-opacity duration-200" />

        <div className="absolute bottom-3 left-3 right-3 flex items-center justify-between">
          <div className={`px-2.5 py-1 text-[10px] font-bold uppercase tracking-wider rounded-full ${
            displayStatus === 'approved'
              ? 'bg-emerald-400 text-emerald-900'
              : displayStatus === 'pending'
              ? 'bg-amber-400 text-amber-900'
              : 'bg-rose-400 text-rose-900'
          }`}>
            {displayStatus === 'approved' ? 'Approved' : displayStatus === 'pending' ? 'Pending' : 'Declined'}
          </div>

          {!isAdmin && (
            <button
              type="button"
              onClick={(e) => { e.preventDefault(); e.stopPropagation() }}
              className="w-8 h-8 flex items-center justify-center shrink-0 bg-white rounded-full text-slate-700 hover:bg-slate-100 transition-all shadow-[2px_2px_0_0_#475569]"
              data-info-trigger
            >
              <Info className="w-4 h-4" />
            </button>
          )}
        </div>
      </div>

      <div className="p-4 flex flex-col flex-1">
        <h2 className="text-sm md:text-sm font-bold text-slate-900 dark:text-white line-clamp-2 mb-1 leading-tight" title={album.name}>
          {album.name}
        </h2>

        {album.school_city && (
          <p className="text-xs text-slate-500 dark:text-slate-400 font-medium truncate">
            {album.school_city}
          </p>
        )}

        <div className="mt-auto pt-3 space-y-2">
          {isApproved && !isPaid && onPay && (
            <button
              type="button"
              disabled={!!loadingId}
              onClick={(e) => { e.preventDefault(); e.stopPropagation(); onPay(album) }}
              className="w-full py-2.5 text-xs font-bold rounded-xl bg-emerald-400 text-emerald-900 hover:bg-emerald-300 transition-colors shadow-[2px_2px_0_0_#059669] disabled:opacity-50"
            >
              Bayar Sekarang
            </button>
          )}

          {album.isOwner !== false && onInvite && (album.album_id ?? album.type === 'public') && (
            <button
              type="button"
              onClick={(e) => { e.preventDefault(); e.stopPropagation(); onInvite(album) }}
              className="w-full py-2.5 text-xs font-bold rounded-xl bg-pink-100 dark:bg-pink-900/50 text-pink-700 dark:text-pink-200 hover:bg-pink-200 dark:hover:bg-pink-900 transition-colors shadow-[2px_2px_0_0_#db2777] flex items-center justify-center gap-2"
            >
              <UserPlus className="w-3.5 h-3.5" />
              Undang Teman
            </button>
          )}

          {album.type === 'yearbook' && isApproved && (isPaid || isAdmin) && editorUrl && (
            <Link
              href={editorUrl}
              onClick={(e) => {
                e.stopPropagation()
                setNavigatingAlbumId?.(album.id)
              }}
              prefetch
              scroll={false}
              className={`w-full py-2.5 text-xs font-bold rounded-xl flex items-center justify-center gap-2 transition-colors shadow-[2px_2px_0_0_#7c3aed] ${
                isNavigatingToEditor
                  ? 'bg-slate-200 dark:bg-slate-700 text-slate-500 dark:text-slate-300 pointer-events-none'
                  : 'bg-violet-400 text-violet-900 hover:bg-violet-300'
              }`}
            >
              {isNavigatingToEditor ? (
                <>
                  <Loader2 className="w-3.5 h-3.5 animate-spin" />
                  Membuka...
                </>
              ) : (
                <>
                  <LayoutDashboard className="w-3.5 h-3.5" />
                  Edit Album
                </>
              )}
            </Link>
          )}
        </div>

        {!isAdmin && (
          <p className="text-[11px] text-slate-400 dark:text-slate-500 text-center mt-3">
            {isClickable ? 'Klik untuk membuka' : statusLabel === 'pending' ? 'Menunggu persetujuan' : 'Selesaikan pembayaran'}
          </p>
        )}
      </div>
    </div>
  )

  return <CardContent />
}

function AlbumInfoModal({
  album,
  onClose
}: {
  album: AlbumRow
  onClose: () => void
}) {
  const isPaid = album.isOwner !== false ? album.payment_status === 'paid' : (album.individual_payments_enabled ? album.member_payment_status === 'paid' : album.payment_status === 'paid')
  const created = album.created_at ? new Date(album.created_at).toLocaleDateString('id-ID', { day: 'numeric', month: 'long', year: 'numeric' }) : null
  const estimasi = album.total_estimated_price
    ? new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', maximumFractionDigits: 0 }).format(album.total_estimated_price)
    : null

  return (
    <div
      className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-slate-900/40 backdrop-blur-sm animate-in fade-in duration-200"
      onClick={onClose}
    >
      <div
        className="bg-white dark:bg-slate-900 rounded-3xl w-full max-w-md shadow-[4px_4px_0_0_#334155] dark:shadow-[4px_4px_0_0_#475569] animate-in zoom-in-95 duration-200"
        onClick={(e) => e.stopPropagation()}
      >
        <div className="p-5">
          <div className="flex items-center justify-between mb-5">
            <h2 className="text-lg font-bold text-slate-900 dark:text-white">Informasi Album</h2>
            <button
              type="button"
              onClick={onClose}
              className="p-2 rounded-xl bg-slate-100 dark:bg-slate-800 text-slate-500 hover:text-slate-700 dark:hover:text-slate-200 hover:bg-slate-200 dark:hover:bg-slate-700 transition-all"
            >
              <X className="w-5 h-5" />
            </button>
          </div>

          <p className="text-base font-bold text-slate-900 dark:text-white mb-1 line-clamp-2 leading-snug">
            {album.name}
          </p>
          {album.description && (
            <p className="text-sm text-slate-500 dark:text-slate-400 mb-5 line-clamp-2">
              {album.description}
            </p>
          )}

          <div className="space-y-3">
            <div className="flex items-center gap-3 p-3 bg-emerald-50 dark:bg-emerald-950/30 rounded-2xl">
              <div className="w-10 h-10 rounded-xl bg-emerald-200 dark:bg-emerald-900/50 flex items-center justify-center shrink-0">
                <ShieldCheck className="w-5 h-5 text-emerald-600 dark:text-emerald-400" />
              </div>
              <div className="flex-1 min-w-0">
                <p className="text-[10px] font-semibold text-emerald-600 dark:text-emerald-400 uppercase tracking-wider">Status</p>
                <p className={`text-sm font-bold ${album.status === 'approved' ? 'text-emerald-700 dark:text-emerald-300' : 'text-amber-700 dark:text-amber-300'}`}>
                  {album.status === 'approved' ? 'Approved' : album.status === 'pending' ? 'Pending' : 'Declined'}
                </p>
              </div>
            </div>

            <div className={`flex items-center gap-3 p-3 rounded-2xl ${isPaid ? 'bg-violet-50 dark:bg-violet-950/30' : 'bg-rose-50 dark:bg-rose-950/30'}`}>
              <div className={`w-10 h-10 rounded-xl flex items-center justify-center shrink-0 ${isPaid ? 'bg-violet-200 dark:bg-violet-900/50' : 'bg-rose-200 dark:bg-rose-900/50'}`}>
                <CreditCard className={`w-5 h-5 ${isPaid ? 'text-violet-600 dark:text-violet-400' : 'text-rose-500 dark:text-rose-400'}`} />
              </div>
              <div className="flex-1 min-w-0">
                <p className={`text-[10px] font-semibold uppercase tracking-wider ${isPaid ? 'text-violet-600 dark:text-violet-400' : 'text-rose-500 dark:text-rose-400'}`}>Pembayaran</p>
                <p className={`text-sm font-bold ${isPaid ? 'text-violet-700 dark:text-violet-300' : 'text-rose-700 dark:text-rose-300'}`}>
                  {isPaid ? 'Lunas' : 'Belum Bayar'}
                </p>
              </div>
            </div>

            <div className="flex items-center gap-3 p-3 bg-amber-50 dark:bg-amber-950/30 rounded-2xl">
              <div className="w-10 h-10 rounded-xl bg-amber-200 dark:bg-amber-900/50 flex items-center justify-center shrink-0">
                <Package className="w-5 h-5 text-amber-600 dark:text-amber-400" />
              </div>
              <div className="flex-1 min-w-0">
                <p className="text-[10px] font-semibold text-amber-600 dark:text-amber-400 uppercase tracking-wider">Paket</p>
                <p className="text-sm font-bold text-slate-900 dark:text-white truncate">
                  {album.type === 'yearbook'
                    ? (album.pricing_packages?.name?.replace(/^Paket\s+/i, '') || 'Yearbook')
                    : 'Shared Gallery'}
                </p>
              </div>
            </div>

            <div className="flex items-center gap-3 p-3 bg-sky-50 dark:bg-sky-950/30 rounded-2xl">
              <div className="w-10 h-10 rounded-xl bg-sky-200 dark:bg-sky-900/50 flex items-center justify-center shrink-0">
                <Calendar className="w-5 h-5 text-sky-600 dark:text-sky-400" />
              </div>
              <div className="flex-1 min-w-0">
                <p className="text-[10px] font-semibold text-sky-600 dark:text-sky-400 uppercase tracking-wider">Tanggal</p>
                <p className="text-sm font-bold text-slate-900 dark:text-white">{created || '-'}</p>
              </div>
            </div>

            <div className="flex items-center gap-3 p-3 bg-pink-50 dark:bg-pink-950/30 rounded-2xl">
              <div className="w-10 h-10 rounded-xl bg-pink-200 dark:bg-pink-900/50 flex items-center justify-center shrink-0">
                <User className="w-5 h-5 text-pink-600 dark:text-pink-400" />
              </div>
              <div className="flex-1 min-w-0">
                <p className="text-[10px] font-semibold text-pink-600 dark:text-pink-400 uppercase tracking-wider">Pembuat</p>
                <p className="text-sm font-bold text-slate-900 dark:text-white truncate">{album.pic_name || 'Tidak diketahui'}</p>
              </div>
            </div>

            {album.school_city && (
              <div className="flex items-center gap-3 p-3 bg-cyan-50 dark:bg-cyan-950/30 rounded-2xl">
                <div className="w-10 h-10 rounded-xl bg-cyan-200 dark:bg-cyan-900/50 flex items-center justify-center shrink-0">
                  <Package className="w-5 h-5 text-cyan-600 dark:text-cyan-400" />
                </div>
                <div className="flex-1 min-w-0">
                  <p className="text-[10px] font-semibold text-cyan-600 dark:text-cyan-400 uppercase tracking-wider">Kota</p>
                  <p className="text-sm font-bold text-slate-900 dark:text-white">{album.school_city}</p>
                </div>
              </div>
            )}

            {estimasi && (
              <div className="flex items-center gap-3 p-3 bg-orange-50 dark:bg-orange-950/30 rounded-2xl">
                <div className="w-10 h-10 rounded-xl bg-orange-200 dark:bg-orange-900/50 flex items-center justify-center shrink-0">
                  <CreditCard className="w-5 h-5 text-orange-600 dark:text-orange-400" />
                </div>
                <div className="flex-1 min-w-0">
                  <p className="text-[10px] font-semibold text-orange-600 dark:text-orange-400 uppercase tracking-wider">Estimasi</p>
                  <p className="text-sm font-bold text-slate-900 dark:text-white">{estimasi}</p>
                </div>
              </div>
            )}
          </div>
        </div>
      </div>
    </div>
  )
}
